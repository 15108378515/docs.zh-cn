```csharp  
   public int count;  
   // try the following line instead  
   // public static int count;  
  
   class InnerClass  
   {  
      void func()  
      {  
         // or, create an instance  
         // OuterClass class_inst = new OuterClass();  
         // int count2 = class_inst.count;

      }  
   }  
  
   public static void Main()  
   {  
   }  
}  
```
