## <a name="example"></a><span data-ttu-id="80610-111">示例</span><span class="sxs-lookup"><span data-stu-id="80610-111">Example</span></span>  
 <span data-ttu-id="80610-112">以下代码生成 CS0313：</span><span class="sxs-lookup"><span data-stu-id="80610-112">The following code generates CS0313:</span></span>  
  
```  
// cs0313.cs  
public interface BaseInterface { }  
public struct ImplStruct : BaseInterface { }  
  
public class TestClass  
{  
    public T? TestMethod<T, U>(T t) where T : struct, U  
    {  
        return t;  
    }  
}  
  
public class NullableTest  
{  
    public static void Run()  
    {  
  
        TestClass tc = new TestClass();  
        tc.TestMethod<ImplStruct?, BaseInterface>(new ImplStruct?()); // CS0313  
    }  
    public static void Main()  
    { }  
}  
```
