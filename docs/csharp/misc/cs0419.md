```csharp  
// cs0419.cs

{  
   /// text for F(void)  
   void F();  
   /// text for F(int)  
   void F(int i);  
}  
/// text for class MyClass

{  
   /// <see cref="I.F"/>  
   public static void MyMethod(int i)  
   {  
   }

   /// <see cref="I.F(int)"/>  
   public static void MyMethod(int i)  
   {  
   }  
*/  
   /// text for Main  
   public static void Main ()  
   {  
   }  
}  
```
