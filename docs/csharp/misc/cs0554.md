 <span data-ttu-id="5a7e2-104">不允许用户定义的转换转换为派生类的值；不需要此类运算符。</span><span class="sxs-lookup"><span data-stu-id="5a7e2-104">User-defined conversions to values of a derived class are not allowed; you do not need such an operator.</span></span>

```csharp  
// CS0554.cs  
namespace x  
{  
   public class ii  
   {  
      // delete the conversion routine to resolve CS0554  
      public static implicit operator ii(a aa) // CS0554  
      {  
         return new ii();  
      }  
   }  
  
   public class a : ii  
   {  
      public static void Main()  
      {  
      }  
   }  
}  
```
